Mysql表的操作

2022/10/12 数据库 linux mysql

一个数据库由多个表组成

1、创建表

CREATE TABLE 表名称(字段名称 类型，…)

create table t_user(
id int primary key auto_increment,
username varchar(10) not null comment "用户名称",
create_time date not null comment "最近登录时间"
)

主键

primary key

create table t_user(
id int primary key auto_increment,
)

外键

添加

alter 表名称 add constraint 外键名称 foreign key (外键) references 参考表(参考字段);

alter t_user_role add constraint u_r_fk foreign key (uid) references t_user(id);

删除外键

alter table 表名称 drop foreign key 外键名称;

alter table t_user_role drop foreign key u_r_fk;

2、修改表

修改表名

alter table t_user rename to t_users;

修改表字段

alter table t_user modify username varchar(20)

删除表字段

alter table t_user drop create_time

添加表字段

alter table t_user add password varchar(15) not null;

3、删除表

drop table t_users;

4、查询

查询表结构方式有两种：desc和show create table <表名称>

desc t_users;
show create table t_users;
